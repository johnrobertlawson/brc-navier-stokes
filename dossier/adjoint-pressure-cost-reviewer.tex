\paragraph{Scope.}
This self-contained note gives the exact finite-window
adjoint-pressure functional used in the argument and the proved
a priori lower bound for one Besov regeneration event.  The statement is
conditional on the existence of the smooth physical genealogy and the
event sequence specified below.  It neither sums the costs over infinitely
many events nor proves regularity or a resolution of the Clay problem.

\section*{1. Smooth genealogy and its Oseen adjoint}

Fix viscosity \(\nu>0\).  Let \(X\) be an ancient trajectory state with
terminal trace \(g=\ev_0X\).  An admissible smooth physical genealogy for
\(X\) is a sequence
\[
  \cG=\{(u_n,H_n)\}_{n\ge1}
\]
such that:
\begin{enumerate}
  \item \(u_n\) is a smooth, finite-energy, divergence-free
        Navier--Stokes solution on
        \(\R^3\times[-H_n,0]\), with viscosity \(\nu\);
  \item \(H_n\to\infty\);
  \item for one \(M>0\),
        \begin{equation}
          \sup_n\sup_{-H_n\le t\le0}
          \|u_n(t)\|_{\weakLthree(\R^3)}
          \le M;
          \label{eq:weak-L3}
        \end{equation}
  \item \(u_n\to X\) strongly in local spacetime \(L^3\) on every
        compact negative-time cylinder, with the corresponding local weak
        pressure convergence; and
  \item \(u_n(0)\to g\) in \(\mathcal D'(\R^3)\).
\end{enumerate}

For \(\psi\in C^\infty_{c,\sigma}(\R^3;\R^3)\), set
\(b_n(\tau)=u_n(-\tau)\).  For \(0<T\le H_n\), let
\((a_{n,\psi},\pi^*_{n,\psi})\) solve
\begin{equation}
  \boxed{
  \begin{aligned}
    \partial_\tau a_{n,\psi}
      -\nu\Delta a_{n,\psi}
      -b_n\cdot\nabla a_{n,\psi}
      +\nabla\pi^*_{n,\psi}&=0,
      &&0<\tau<T,\\
    \nabla\cdot a_{n,\psi}&=0,\\
    a_{n,\psi}(0)&=\psi .
  \end{aligned}}
  \label{eq:adjoint}
\end{equation}
Equivalently,
\(\partial_\tau a_{n,\psi}-\nu\Delta a_{n,\psi}
-\mathbb P(b_n\cdot\nabla a_{n,\psi})=0\), where
\(\mathbb P\) is the Leray projection.

\section*{2. Finite-horizon estimate and exact cost}

\begin{proposition}[smooth finite-horizon pressure estimate]
\label{prop:finite-horizon}
There is a universal constant \(C_{\rm adj}>0\), depending only on the
normalisations of Nash's inequality and Lorentz duality, such that
\begin{equation}
  \boxed{
  \int_0^T
  \|\nabla\pi^*_{n,\psi}(\tau)\|_{L^1(\R^3)}\,d\tau
  \ge
  \frac{|\langle u_n(0),\psi\rangle|}
       {C_{\rm adj}M}\sqrt{\nu T}
  -\|\psi\|_{L^1(\R^3)} .}
  \label{eq:finite-horizon}
\end{equation}
\end{proposition}

\begin{proof}
In reversed time the primal equation and \eqref{eq:adjoint} have the
conserved pairing and adjoint energy identity
\begin{align}
  \langle b_n(T),a_{n,\psi}(T)\rangle
    &=\langle u_n(0),\psi\rangle,
    \label{eq:pairing}\\
  \|a_{n,\psi}(T)\|_2^2
  +2\nu\int_0^T\|\nabla a_{n,\psi}(\tau)\|_2^2\,d\tau
    &=\|\psi\|_2^2.
    \label{eq:energy}
\end{align}
Put
\[
  B_n(T):=\sup_{0\le\tau\le T}\|a_{n,\psi}(\tau)\|_1.
\]
Nash's inequality, \eqref{eq:energy}, and real interpolation give
\[
  \|a_{n,\psi}(T)\|_{\LorentzDual}
  \le C_* B_n(T)(\nu T)^{-1/2}.
\]
By \eqref{eq:weak-L3}, \eqref{eq:pairing}, and Lorentz duality,
\[
  |\langle u_n(0),\psi\rangle|
  \le C_{\rm Lor}M
      \|a_{n,\psi}(T)\|_{\LorentzDual}.
\]
Thus, with \(C_{\rm adj}=C_{\rm Lor}C_*\),
\[
  B_n(T)
  \ge
  \frac{|\langle u_n(0),\psi\rangle|}
       {C_{\rm adj}M}\sqrt{\nu T}.
\]
Finally, the vector Kato inequality for \eqref{eq:adjoint} gives
\[
  \frac{d}{d\tau}\|a_{n,\psi}(\tau)\|_1
  \le\|\nabla\pi^*_{n,\psi}(\tau)\|_1.
\]
Taking the running supremum and using
\(\|a_{n,\psi}(0)\|_1=\|\psi\|_1\) proves
\eqref{eq:finite-horizon}.  If either \(B_n(T)\) or the pressure integral
is infinite, the corresponding inequality holds in the extended sense.
\end{proof}

\paragraph{Definition of the finite-window cost.}
For \(T>0\), define
\begin{equation}
  \boxed{
  \mathfrak p^\cG_{\psi,T}(X)
  :=
  \liminf_{\substack{n\to\infty\\H_n\ge T}}
  \frac{1}{\sqrt{\nu T}}
  \int_0^T
  \|\nabla\pi^*_{n,\psi}(\tau)\|_{L^1(\R^3)}\,d\tau .}
  \label{eq:cost}
\end{equation}
The constrained lower limit means
\[
  \lim_{N\to\infty}
  \inf_{\substack{n\ge N\\H_n\ge T}}(\cdots).
\]
It is taken only along the smooth genealogy; no endpoint adjoint on the
rough state \(X\) is assumed.

Dividing \eqref{eq:finite-horizon} by \(\sqrt{\nu T}\), using
\(u_n(0)\to\ev_0X\) distributionally, and taking the lower limit yields
\begin{equation}
  \boxed{
  \mathfrak p^\cG_{\psi,T}(X)
  \ge
  \frac{|\langle\ev_0X,\psi\rangle|}
       {C_{\rm adj}M}
  -\frac{\|\psi\|_1}{\sqrt{\nu T}} .}
  \label{eq:cost-lower}
\end{equation}

\section*{3. One Besov regeneration event}

From now on take \(X=X_*\), where \(X_*\) is the ancient profile, and let
\(\cG\) be the admissible genealogy for \(X_*\) fixed in Section~1.  Let
the Navier--Stokes dilation be
\[
  (\cS_\lambda u)(x,t)
  :=\lambda u(\lambda x,\lambda^2t),
  \qquad
  \Phi_\theta:=\cS_{e^\theta}.
\]
For a fixed \(\lambda>0\), define the scaled genealogy, including its
available backward horizons, by
\begin{equation}
  \cS_\lambda\cG
  :=
  \left\{
    \left(\cS_\lambda u_n,\frac{H_n}{\lambda^2}\right)
  \right\}_{n\ge1}.
  \label{eq:scaled-genealogy}
\end{equation}
Choose
\(\varphi\in C^\infty_{c,\sigma}(\R^3;\R^3)\), and define the detector
\begin{equation}
  z(\theta)
  :=
  \left\langle
    \ev_0(\Phi_\theta X_*),\varphi
  \right\rangle .
  \label{eq:observable}
\end{equation}
Assume there are regeneration times \(\theta_m\) and a \(c_0>0\) such
that
\[
  z(\theta_m)\ge c_0,
  \qquad
  \theta_{m+1}-\theta_m\ge1.
\]
Set
\[
  \mathcal A\varphi:=2\varphi+x\cdot\nabla\varphi,
  \qquad
  L_\varphi
  :=C_{\rm Lor}M
    \|\mathcal A\varphi\|_{\LorentzDual}.
\]
The exact dilation derivative and Lorentz duality give
\[
  z'(\theta)
  =-\left\langle
      \ev_0(\Phi_\theta X_*),\mathcal A\varphi
    \right\rangle,
  \qquad
  |z'(\theta)|\le L_\varphi.
\]
Consequently, for
\begin{equation}
  \delta
  :=
  \begin{cases}
    \displaystyle
    \min\left\{\frac12,\frac{c_0}{2L_\varphi}\right\},
      &L_\varphi>0,\\[6pt]
    \displaystyle\frac12,&L_\varphi=0,
  \end{cases}
  \label{eq:delta}
\end{equation}
every event has the forward half-height interval
\begin{equation}
  z(\theta)\ge\frac{c_0}{2}
  \qquad
  (\theta_m\le\theta\le\theta_m+\delta).
  \label{eq:persistence}
\end{equation}

Choose one finite horizon \(T_0\) satisfying
\begin{equation}
  \boxed{
  \sqrt{\nu T_0}
  \ge
  \frac{4C_{\rm adj}M\|\varphi\|_1}{c_0}.}
  \label{eq:horizon}
\end{equation}
For each fixed \(\theta\), use the correspondingly scaled genealogy
\(\cG_\theta:=\cS_{e^\theta}\cG\).

\begin{corollary}[single-event packet bound]
\label{cor:single-event}
On the active interval of the \(m\)-th regeneration event,
\begin{equation}
  \boxed{
  \mathfrak p^{\cG_\theta}_{\varphi,T_0}
    (\Phi_\theta X_*)
  \ge
  \frac{c_0}{4C_{\rm adj}M}
  =:\beta_0
  \qquad
  (\theta_m\le\theta\le\theta_m+\delta).}
  \label{eq:pointwise-packet}
\end{equation}
In particular, define its finite adjoint-pressure packet by
\begin{equation}
  \boxed{
  \mathcal P_m
  :=
  \int_{\theta_m}^{\theta_m+\delta}{}^{\!*}
  \mathfrak p^{\cG_\theta}_{\varphi,T_0}
    (\Phi_\theta X_*)\,d\theta,}
  \label{eq:packet}
\end{equation}
where, for a nonnegative function \(F\), the lower Lebesgue integral is
\[
  \int_E^{\!*}F
  :=
  \sup\left\{
    \int_E h:
    h\ \text{is measurable and }0\le h\le F
  \right\}.
\]
Then the a priori cost of one regeneration event obeys
\begin{equation}
  \boxed{
  \mathcal P_m
  \ge
  \delta\beta_0
  =
  \frac{\delta c_0}{4C_{\rm adj}M}
  >0.}
  \label{eq:single-event}
\end{equation}
\end{corollary}

\begin{proof}
For \(\theta\) in \eqref{eq:persistence}, apply
\eqref{eq:cost-lower} with \(X=\Phi_\theta X_*\),
\(\psi=\varphi\), and \(T=T_0\).  The first term is at least
\(c_0/(2C_{\rm adj}M)\), while \eqref{eq:horizon} bounds the penalty by
\(c_0/(4C_{\rm adj}M)\).  This proves
\eqref{eq:pointwise-packet}.  The definition of the lower integral then
gives \eqref{eq:single-event}.
\end{proof}

\paragraph{Scale check.}
For
\(\psi_\lambda(x)=\lambda^{-2}\psi(x/\lambda)\), the exact covariance is
\[
  \boxed{
  \mathfrak p^{\cS_\lambda\cG}_{\psi,T}
    (\cS_\lambda X)
  =
  \mathfrak p^\cG_{\psi_\lambda,\lambda^2T}(X).}
\]
Thus the packet is scale zero after division by \(\sqrt{\nu T}\), and
the fixed scaled horizon \(T_0\) corresponds to the finite physical
horizon \(\lambda^2T_0\).

\paragraph{Open boundary.}
Equation \eqref{eq:single-event} is a positive lower bound for one event.
It does not bound \(\sum_m\mathcal P_m\), because the adjoint histories
at different scales can overlap or concentrate in terminal initial
layers.  Excluding those mechanisms requires an additional theorem.
